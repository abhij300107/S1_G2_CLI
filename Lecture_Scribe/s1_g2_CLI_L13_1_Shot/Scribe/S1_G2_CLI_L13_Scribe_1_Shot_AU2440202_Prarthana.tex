\section*{Q1}

\textbf{(a) Total number of ways to divide 20 distinct dishes into four groups of five}

\textbf{Concept Used:}
\begin{itemize}
    \item Combinations for selection without order
    \item Division into identical groups
\end{itemize}

\textbf{Formula Used:}
\[
\frac{1}{4!}
\binom{20}{5}
\binom{15}{5}
\binom{10}{5}
\binom{5}{5}
\]

\textbf{Notation:}
\[
\binom{n}{k} : \text{Number of ways of selecting } k \text{ objects from } n
\]
\[
4! : \text{Removes ordering among groups}
\]

\textbf{Why Used:}
\begin{itemize}
    \item Dishes are distinct $\Rightarrow$ use combinations
    \item Groups are unordered $\Rightarrow$ divide by $4!$
\end{itemize}

\vspace{0.3cm}

\textbf{(b) Probability that all four platters contain dishes of the same cuisine}

\textbf{Concept Used:}
\[
\text{Probability} = \frac{\text{Favorable outcomes}}{\text{Total outcomes}}
\]

\textbf{Given:}
\begin{itemize}
    \item Equal number of dishes per cuisine
    \item Each cuisine contains 5 dishes
\end{itemize}

\textbf{Why Used:} Event probability computed as ratio of favorable arrangements to total arrangements.

\section*{Q2}

A passenger arrives uniformly between 7:00 and 7:30.

\textbf{Concept Used: Continuous Uniform Distribution}

\[
X \sim \text{Uniform}(a,b)
\]

\[
f_X(x) = \frac{1}{b-a}, \quad a \le x \le b
\]

\textbf{Notation:}
\[
a = 0, \quad b = 30
\]

\textbf{(a) Wait less than 5 minutes}
\[
P(\text{Wait} < 5)
\]

\textbf{(b) Wait more than 10 minutes}
\[
P(\text{Wait} > 10)
\]

\section*{Q3}

\[
P(\text{Late}) = 0.15
\]
\[
P(\text{Early}) = 0.25
\]
\[
P(\text{Late} \cap \text{Early}) = 0.08
\]

\textbf{Find:}
\[
P(\text{Early} \mid \text{Late})
\]

\textbf{Concept Used: Conditional Probability}

\[
P(A|B) = \frac{P(A \cap B)}{P(B)}
\]

\[
P(X=k) = \frac{e^{-\lambda} \lambda^k}{k!}

\[
P(X=0) = e^{-0.2}

\[
P(X \ge 2) = 1 - P(X=0) - P(X=1)

\section*{Q9}

\[
p(i) = c \frac{\lambda^i}{i!}, \quad i=0,1,2,\dots
\]

\textbf{Normalization Condition:}
\[
\sum_{i=0}^{\infty} p(i) = 1
\]

\[
c \sum_{i=0}^{\infty} \frac{\lambda^i}{i!} = 1
\]

\[
c e^{\lambda} = 1
\]

\[
c = e^{-\lambda}
\]

\textbf{(a)}
\[
P(X=0) = e^{-\lambda}
\]

\textbf{(b)}
\[
P(X>2) = 1 - P(X=0) - P(X=1) - P(X=2)
\]
